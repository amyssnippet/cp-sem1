{
static int n1=0,n2=1,n3;

{
n3 = n1 + n2;
n1 = n2;
n2 = n3;

}

EXP 8

Write a C program that defines a structure called student , store the student's name,roll no,address,date of birth and print it

#include < stdio.h>
#include < string.h>
struct student {
char name[100];
int roll_no;
char address[200];
char dob[20];
};
int main() {
struct student stu;
printf("Enter student's name: ");
fgets(stu.name, sizeof(stu.name), stdin);
stu.name[strcspn(stu.name, "\n")] = '\0';
printf("Enter student's roll number: ");
scanf("%d", &stu.roll_no);
getchar();
printf("Enter student's address: ");
fgets(stu.address, sizeof(stu.address), stdin);
stu.address[strcspn(stu.address, "\n")] = '\0';
printf("Enter student's date of birth (DD/MM/YYYY): ");
fgets(stu.dob, sizeof(stu.dob), stdin);
stu.dob[strcspn(stu.dob, "\n")] = '\0';
printf("\nStudent Information:\n");
printf("Name: %s\n", stu.name);
printf("Roll No: %d\n", stu.roll_no);
printf("Address: %s\n", stu.address);
printf("Date of Birth: %s\n", stu.dob);
return 0;
}
